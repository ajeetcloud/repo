/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public void flatten(TreeNode root) {
        
        TreeNode head = new TreeNode();
        dfs(root, head);
        return head;
    }

    private void dfs(TreeNode root, TreeNode head) {

        if (root == null) {
            return;
        }
        // visit
        head.val = root.val;
        head.left = null;

        TreeNode nextRight = new TreeNode()
        head.right = nextRight;

        dfs(root.left, head.right);
        dfs(root.right, head.right);

    }
}
