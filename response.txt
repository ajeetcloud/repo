class Solution {

    int maxDepth = 0;

    public int maxDepth(TreeNode root) {
        
        dfsDepth(root);
    }

    private int dfsDepth(TreeNode node) {

        if (node == null) {
            return 0;
        }
        int leftDepth = 1 + dfsDepth(node.left);
        int rightDepth = 1 + dfsDepth(node.right);

        maxDepth = Math.max(maxDepth, Math.max(leftDepth, rightDepth));

        return Math.max(leftDepth, rightDepth);
    }
}
