class Solution {
    public int kthSmallest(TreeNode root, int k) {

        int dfs(root, k);
        
    }

    // Inorder
    private int dfs(TreeNode root, int k) {

        if (root == null) {
            return -1;
        }
        if (k == 0) {
            return root.val;
        }
        dfs(root.left, k);
        // visit root
        k--;
        if (k == 0) {
            return root.val;
        }
        dfs(root.right, k);
    }
}
