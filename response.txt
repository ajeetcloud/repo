    }

    private int dfs(Node root) {

        int best1 = 0;
        int best2 = 0;

        if (root == null) {
            return 0;
        }

        List<Node> children = root.children;

        for (int child: children) {

            int childHeight = dfs(child);
            if (childHeight > best1) {
                best2 = best1;
                best1 = childHeight;
            }
            else if (childHeight > best2) {
                best2 = childHeight;
            }
        }

        return 1 + best1;
    }
