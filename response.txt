class Solution {
    public Node connect(Node root) {

        Node leftmost = root;

        while (leftmost != null) {

            Node firstNodeOfLevel = null;
            Node tail = null;

            // Current level Sweep
            Node head = leftmost;

            while (head != null) {
                if (head.left != null) {
                    if (firstNodeOfLevel == null) {
                        firstNodeOfLevel = head.left;
                    } else {
                        tail.next = head.next;
                    }
                    tail = head.left;
                }
                if (head.right != null) {
                    if (firstNodeOfLevel == null) {
                        firstNodeOfLevel = head.right;
                    } else {
                        tail.next = head.right;
                    }
                    tail = head.right;
                }
                head = head.next;
            }
            // starts next level
            leftmost = firstNodeOfLevel;
        }
        return root;
    }
}
