class Solution {
    public Node connect(Node root) {

        Node leftmost = root;

        while (leftmost != null) {
            Node firstNode = null;
            Node tail = null;

            Node head = leftmost;
            while (head != null) {
                if (head.left != null) {
                    if (firstNode == null) {
                        firstNode = head.left;
                        tail = head.left;
                    } else {
                        tail.next = head.left;
                    }
                    tail = tail.next;
                }

                if (head.right != null) {
                    if (firstNode == null) {
                        firstNode = head.right;
                        tail = head.right;
                    } else {
                        tail.next = head.right;
                    }
                    tail = tail.next;
                }
                head = head.next;
            }
            leftmost = firstNode;
        }
        return root;
    }
}
