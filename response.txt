class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        
        return dfsSum(root, targetSum);
    }

    private boolean dfsSum(TreeNode root, int targetSum) {

        if (root == null) {
            if (targetSum == 0) {
                return true;
            }
            return false;
        }

        return dfsSum(root.left, targetSum - root.val) && dfsSum(root.right, targetSum - root.val);
    }
}
